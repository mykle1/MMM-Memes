## Config.js entry and options

{
          disabled: f,
 module: "MMM-Memes",
 position: "bottom right",
 config: {
   useHeader: false,                            // false if you don't want a header
   header: "",                                 // Any text you want. useHeader must be true
   maxWidth: "300px",
   animationSpeed: 3000,                       // fade speed
   rotateInterval: 5 * 60 * 1000,
 }
},
